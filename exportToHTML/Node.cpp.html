#include <iostream>
#include <string>
using namespace std;

template <typename T>
class Node {
public:
    T& eventNum;
    Node* left;
    Node* right;
    string& description;

    Node(T& d, string& ds): eventNum(d), left(nullptr), right(nullptr), description(ds) {}
    void print() { cout << description << " ";}
};
